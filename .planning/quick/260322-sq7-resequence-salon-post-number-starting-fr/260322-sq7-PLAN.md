    Append the following route to vendorAdmin.js BEFORE the final `export default router` line (keep it near the other maintenance-style routes such as `/delete-salon`).

    The route MUST:
    - Use `requireSecret` and `requirePin` middleware — same pattern as every other internal POST
    - Accept `?salon=<slug>` from req.query
    - Return 400 JSON `{ error: "salon query param required" }` if salon is missing
    - Fetch all posts WHERE salon_id = salon AND status NOT IN ('cancelled', 'draft') ORDER BY COALESCE(published_at, scheduled_for) ASC, id ASC
    - Use a single `db.transaction()` to UPDATE each post's salon_post_number to its 1-based position in the ordered list
    - Return 200 JSON: `{ salon, updated: N, mappings: [ { id, old: N, new: N }, ... ] }`
    - Wrap everything in try/catch and return 500 JSON `{ error: message }` on failure

    Example skeleton (implement exactly — do NOT change auth middleware or response shape):

    ```js
    router.post("/resequence-posts", requireSecret, requirePin, (req, res) => {
      const { salon } = req.query;
      if (!salon) return res.status(400).json({ error: "salon query param required" });

      try {
        const posts = db.prepare(`
          SELECT id, salon_post_number
          FROM posts
          WHERE salon_id = ?
            AND status NOT IN ('cancelled', 'draft')
          ORDER BY COALESCE(published_at, scheduled_for) ASC, id ASC
        `).all(salon);

        if (posts.length === 0) {
          return res.json({ salon, updated: 0, mappings: [] });
        }

        const mappings = [];
        const update = db.prepare(
          "UPDATE posts SET salon_post_number = ? WHERE id = ?"
        );

        const resequence = db.transaction(() => {
          posts.forEach((post, idx) => {
            const newNum = idx + 1;
            mappings.push({ id: post.id, old: post.salon_post_number, new: newNum });
            update.run(newNum, post.id);
          });
        });

        resequence();

        return res.json({ salon, updated: posts.length, mappings });
      } catch (err) {
        return res.status(500).json({ error: err.message });
      }
    });
    ```
